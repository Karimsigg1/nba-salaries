    <h1>NBA Team Rosters</h1>

  <div class="controls">

    <input type="text" id="search" placeholder="Search team or player..." oninput="filterCards()" />

  </div>

  <div class="count" id="count"></div>

  <div id="output" class="loading">Loading live data...</div>

  <script>

    let allTeams = [];

    async function loadData() {

      try {

        const res = await fetch('/api/salaries');

        const data = await res.json();

        if (data.error) throw new Error(data.error);

        allTeams = data.teams;

        render(allTeams);

      } catch (e) {

        document.getElementById('output').innerHTML = '<div class="error">Failed to load: ' + e.message + '</div>';

      }

    }

    function render(teams) {

      document.getElementById('count').textContent = teams.length + ' teams';

      if (!teams.length) { document.getElementById('output').innerHTML = '<div class="loading">No results.</div>'; return; }

      document.getElementById('output').className = 'grid';

      document.getElementById('output').innerHTML = teams.map(t => '<div class="card"><div class="card-header">' + (t.logo ? '<img src="' + t.logo + '" />' : '🏀') + '<h2>' + t.team + '</h2><span>' + t.abbreviation + ' · ' + t.players.length + ' players</span></div><table><thead><tr><th>#</th><th>Player</th><th>Pos</th></tr></thead><tbody>' + t.players.map(p => '<tr><td style="color:#475569">' + p.jersey + '</td><td>' + p.name + '</td><td class="pos">' + p.position + '</td></tr>').join('') + '</tbody></table></div>').join('');

    }

    function filterCards() {

      const q = document.getElementById('search').value.toLowerCase();

      if (!q) { render(allTeams); return; }

      const filtered = allTeams.map(t => ({...t, players: t.players.filter(p => p.name.toLowerCase().includes(q))})).filter(t => t.team.toLowerCase().includes(q) || t.players.length > 0);

      render(filtered);

    }

    loadData();

  </script>
